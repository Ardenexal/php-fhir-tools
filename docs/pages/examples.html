Patient Resource Creation

Simple Patient

Create a basic patient resource with minimal required fields:

use Ardenexal\Component\Models\R4\Resource\Patient\FHIRPatient;
use Ardenexal\Component\Models\R4\DataType\FHIRHumanName;

$patient = new FHIRPatient();
$patient->setId('example-simple');

$name = new FHIRHumanName();
$name->setFamily('Smith');
$name->setGiven(['John']);
$patient->addName($name);

$patient->setGender('male');
$patient->setBirthDate('1974-12-25');

// Serialize to JSON
$serializer = new FHIRSerializationService();
$json = $serializer->serializeToJson($patient);
echo $json;

Complex Patient with Full Demographics

Create a patient with complete demographic information:

$patient = new FHIRPatient();
$patient->setId('example-complex');
$patient->setActive(true);

// Add multiple names
$officialName = new FHIRHumanName();
$officialName->setUse('official');
$officialName->setFamily('Smith');
$officialName->setGiven(['John', 'Robert']);
$officialName->setPrefix(['Mr.']);
$patient->addName($officialName);

// Add identifiers
$mrn = new FHIRIdentifier();
$mrn->setSystem('http://hospital.example.org/patients');
$mrn->setValue('MRN-123456');
$mrn->setUse('official');
$patient->addIdentifier($mrn);

// Add telecom
$phone = new FHIRContactPoint();
$phone->setSystem('phone');
$phone->setValue('+1-555-123-4567');
$phone->setUse('home');
$patient->addTelecom($phone);

// Add address
$address = new FHIRAddress();
$address->setUse('home');
$address->setLine(['123 Main St', 'Apt 4B']);
$address->setCity('Springfield');
$address->setState('IL');
$address->setPostalCode('62701');
$address->setCountry('US');
$patient->addAddress($address);

$patient->setGender('male');
$patient->setBirthDate('1974-12-25');

Observation Creation

Blood Pressure Observation

Create a vital signs observation with multiple components:

use Ardenexal\Component\Models\R4\Resource\Observation\FHIRObservation;
use Ardenexal\Component\Models\R4\DataType\FHIRCodeableConcept;
use Ardenexal\Component\Models\R4\DataType\FHIRCoding;
use Ardenexal\Component\Models\R4\Resource\Observation\FHIRObservationComponent;

$observation = new FHIRObservation();
$observation->setId('blood-pressure-example');
$observation->setStatus('final');

// Set category
$category = new FHIRCodeableConcept();
$categoryCoding = new FHIRCoding();
$categoryCoding->setSystem('http://terminology.hl7.org/CodeSystem/observation-category');
$categoryCoding->setCode('vital-signs');
$categoryCoding->setDisplay('Vital Signs');
$category->addCoding($categoryCoding);
$observation->addCategory($category);

// Set code (LOINC for Blood Pressure)
$code = new FHIRCodeableConcept();
$coding = new FHIRCoding();
$coding->setSystem('http://loinc.org');
$coding->setCode('85354-9');
$coding->setDisplay('Blood pressure panel');
$code->addCoding($coding);
$observation->setCode($code);

// Add systolic component
$systolic = new FHIRObservationComponent();
$systolicCode = new FHIRCodeableConcept();
$systolicCoding = new FHIRCoding();
$systolicCoding->setSystem('http://loinc.org');
$systolicCoding->setCode('8480-6');
$systolicCoding->setDisplay('Systolic blood pressure');
$systolicCode->addCoding($systolicCoding);
$systolic->setCode($systolicCode);

$systolicValue = new FHIRQuantity();
$systolicValue->setValue(120);
$systolicValue->setUnit('mmHg');
$systolicValue->setSystem('http://unitsofmeasure.org');
$systolicValue->setCode('mm[Hg]');
$systolic->setValueQuantity($systolicValue);
$observation->addComponent($systolic);

// Add diastolic component
$diastolic = new FHIRObservationComponent();
$diastolicCode = new FHIRCodeableConcept();
$diastolicCoding = new FHIRCoding();
$diastolicCoding->setSystem('http://loinc.org');
$diastolicCoding->setCode('8462-4');
$diastolicCoding->setDisplay('Diastolic blood pressure');
$diastolicCode->addCoding($diastolicCoding);
$diastolic->setCode($diastolicCode);

$diastolicValue = new FHIRQuantity();
$diastolicValue->setValue(80);
$diastolicValue->setUnit('mmHg');
$diastolicValue->setSystem('http://unitsofmeasure.org');
$diastolicValue->setCode('mm[Hg]');
$diastolic->setValueQuantity($diastolicValue);
$observation->addComponent($diastolic);

// Set subject reference
$subject = new FHIRReference();
$subject->setReference('Patient/example-simple');
$observation->setSubject($subject);

// Set effective datetime
$observation->setEffectiveDateTime('2024-01-15T10:30:00Z');

FHIRPath Evaluation

Basic Expression Evaluation

use Ardenexal\Component\FHIRPath\FHIRPathEvaluator;

$evaluator = new FHIRPathEvaluator();

// Get patient name
$result = $evaluator->evaluate($patient, 'Patient.name.family');
// Returns: ['Smith']

// Check if patient is active
$result = $evaluator->evaluate($patient, 'Patient.active');
// Returns: [true]

// Get all phone numbers
$result = $evaluator->evaluate($patient, "Patient.telecom.where(system='phone').value");
// Returns: ['+1-555-123-4567']

// Get patient age (if birthDate exists)
$result = $evaluator->evaluate($patient, 'Patient.birthDate.toInteger()');

// Check if email exists
$result = $evaluator->evaluate($patient, "Patient.telecom.where(system='email').exists()");
// Returns: [false]

Complex FHIRPath Queries

// Get all observations with systolic > 140
$result = $evaluator->evaluate($bundle, 
    "Bundle.entry.resource.ofType(Observation)" .
    ".where(component.where(code.coding.code='8480-6').value > 140)"
);

// Get patient names in specific format
$result = $evaluator->evaluate($patient,
    "Patient.name.where(use='official').select(family + ', ' + given.first())"
);
// Returns: ['Smith, John']

// Count number of addresses
$result = $evaluator->evaluate($patient, 'Patient.address.count()');
// Returns: [1]

// Get all identifiers of a specific system
$result = $evaluator->evaluate($patient,
    "Patient.identifier.where(system='http://hospital.example.org/patients').value"
);
// Returns: ['MRN-123456']

Bundle Creation and Processing

Create Transaction Bundle

use Ardenexal\Component\Models\R4\Resource\Bundle\FHIRBundle;
use Ardenexal\Component\Models\R4\Resource\Bundle\FHIRBundleEntry;
use Ardenexal\Component\Models\R4\Resource\Bundle\FHIRBundleEntryRequest;

$bundle = new FHIRBundle();
$bundle->setType('transaction');

// Add patient entry
$patientEntry = new FHIRBundleEntry();
$patientEntry->setFullUrl('urn:uuid:' . $this->generateUuid());
$patientEntry->setResource($patient);

$patientRequest = new FHIRBundleEntryRequest();
$patientRequest->setMethod('POST');
$patientRequest->setUrl('Patient');
$patientEntry->setRequest($patientRequest);

$bundle->addEntry($patientEntry);

// Add observation entry
$observationEntry = new FHIRBundleEntry();
$observationEntry->setFullUrl('urn:uuid:' . $this->generateUuid());
$observationEntry->setResource($observation);

$observationRequest = new FHIRBundleEntryRequest();
$observationRequest->setMethod('POST');
$observationRequest->setUrl('Observation');
$observationEntry->setRequest($observationRequest);

$bundle->addEntry($observationEntry);

// Serialize bundle
$json = $serializer->serializeToJson($bundle);

Process Bundle Entries

// Iterate through bundle entries
foreach ($bundle->getEntry() as $entry) {
    $resource = $entry->getResource();
    $resourceType = $resource->getResourceType();
    
    echo "Processing {$resourceType}\n";
    
    if ($resourceType === 'Patient') {
        $patient = $resource;
        // Process patient
    } elseif ($resourceType === 'Observation') {
        $observation = $resource;
        // Process observation
    }
}

// Use FHIRPath to filter entries
$patients = $evaluator->evaluate($bundle, 
    "Bundle.entry.resource.ofType(Patient)"
);

$observations = $evaluator->evaluate($bundle,
    "Bundle.entry.resource.ofType(Observation)"
);

Validation

Strict Validation Mode

use Ardenexal\Component\Serialization\FHIRSerializationService;
use Ardenexal\Component\Serialization\ValidationMode;

$serializer = new FHIRSerializationService();

try {
    $patient = $serializer->deserializeFromJson(
        $json,
        FHIRPatient::class,
        ValidationMode::STRICT
    );
    echo "Validation passed!";
} catch (ValidationException $e) {
    // Get OperationOutcome
    $outcome = $e->getOperationOutcome();
    
    foreach ($outcome->getIssue() as $issue) {
        echo "Severity: " . $issue->getSeverity() . "\n";
        echo "Code: " . $issue->getCode() . "\n";
        echo "Details: " . $issue->getDetails()->getText() . "\n";
        echo "Diagnostics: " . $issue->getDiagnostics() . "\n";
        
        if ($issue->getExpression()) {
            echo "Location: " . implode(', ', $issue->getExpression()) . "\n";
        }
    }
}

Lenient Validation Mode

// Lenient mode allows some non-critical errors
$patient = $serializer->deserializeFromJson(
    $json,
    FHIRPatient::class,
    ValidationMode::LENIENT
);

// Check if there were any warnings
if ($serializer->hasWarnings()) {
    $warnings = $serializer->getWarnings();
    foreach ($warnings as $warning) {
        echo "Warning: " . $warning . "\n";
    }
}
